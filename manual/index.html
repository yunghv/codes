# 使用组件库提供的方法

具名导出

import { transform } from 'zc-erp-lib'

方法调用

transform.date(Date.now(), 2)
代码片断












 




import { transform } from 'zc-erp-lib'
export default {
  components: {},
  data() {
    return {
      loading: false,
    }
  },
  computed: {},
  created() {},
  methods: {
    getDateStr() {
      return `当前日期：${transform.date(Date.now(), 2)}`
    },
  },
}

methods 注册调用

getDateStr: transform.date

<div class="container">当前日期：{{getDateStr(Date.now(),2)}}</div>
