# 起步

# 安装 nrm 及配置私有库

查看所有仓库源地址

nrm ls

# shell 执行结果

切换到私有库

带 * 号为当前私有库镜像源

# 组件库安装及引入

在项目根目录执行 npm 安装组件库

npm install -S zc-erp-lib

在项目入口文件 src/init/module.js 引用组件库

import zcLibrary from 'zc-erp-lib'
Vue.use(zcLibrary, { module: '模块名', prepend: '/main/模块名' })

保留核心代码

# 手动安装必要依赖

安装富文本组件相关依赖

npm install -S quill
npm install -S quill-image-drop-module
npm install -S quill-image-resize-module
npm install -S vue-quill-editor

# 手动挂载 插件

打开文件 vue.config 配置文件 src/vue.config.js，文件开始位置引用 webpack 模块

var webpack = require('webpack')

添加插件代码

const quillPlugin = new webpack.ProvidePlugin({
  'window.Quill': 'quill/dist/quill.js',
  Quill: 'quill/dist/quill.js',
})
config.plugins = [...config.plugins, ...[quillPlugin]]
